LEX Program 2: Classify identifiers, int/real numbers, operators, separators

Code (classify.l):
%{
#include <stdio.h>
%}
%%
[a-zA-Z_][a-zA-Z0-9_]*      { printf("IDENTIFIER: %s\n", yytext); }
[0-9]+"."[0-9]+             { printf("REAL: %s\n", yytext); }
[0-9]+                      { printf("INTEGER: %s\n", yytext); }
"=="|"!="|"<="|">="|"+"|"-"|"*"|"/"|"="|"<"|">" { printf("OPERATOR: %s\n", yytext); }
[(){};,]                    { printf("SEPARATOR: %s\n", yytext); }
[ \t\n]+                    ;
.                           { printf("UNKNOWN: %s\n", yytext); }
%%
int yywrap(){ return 1; }
int main(){ yylex(); return 0; }

Example input:
sum=10+20.5;

Linux:
flex classify.l
gcc lex.yy.c -o classify -lfl
./classify < input.txt

Windows:
win_flex classify.l
gcc lex.yy.c -o classify.exe
classify.exe < input.txt
