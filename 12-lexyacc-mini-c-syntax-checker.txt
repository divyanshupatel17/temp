LEX YACC 5: Mini C Language Syntax Checker

Code (mini.l):
%{
#include "y.tab.h"
%}
%%
"int"                   { return INT; }
"return"                { return RETURN; }
[a-zA-Z_][a-zA-Z0-9_]*  { return ID; }
[0-9]+                  { return NUM; }
"(" | ")" | "{" | "}" | ";" | "=" | "+" { return yytext[0]; }
[ \t\n]+                ;
.                       { return yytext[0]; }
%%
int yywrap(){ return 1; }

Code (mini.y):
%{
#include <stdio.h>
int yylex(void);
void yyerror(char *s){ printf("Syntax Error\n"); }
%}
%token INT RETURN ID NUM
%%
P : INT ID '(' ')' '{' S RETURN NUM ';' '}' { printf("Valid Mini C\n"); };
S : ID '=' NUM '+' NUM ';' ;
%%
int main(){ yyparse(); return 0; }

Example input:
int main(){ a=10+20; return 0; }

Linux:
yacc -d mini.y
lex mini.l
gcc y.tab.c lex.yy.c -o mini -ll -ly
./mini

Windows:
win_bison -d mini.y
win_flex mini.l
gcc mini.tab.c lex.yy.c -o mini.exe
mini.exe
