LEX YACC 1: Simple calculator with +-*/() and precedence

Code (calc.l):
%{
#include "y.tab.h"
%}
%%
[0-9]+      { yylval = atoi(yytext); return NUM; }
[ \t]       ;
\n          { return '\n'; }
.           { return yytext[0]; }
%%
int yywrap(){ return 1; }

Code (calc.y):
%{
#include <stdio.h>
int yylex(void);
void yyerror(char *s){ printf("Invalid\n"); }
%}
%token NUM
%left '+' '-'
%left '*' '/'
%%
S : E '\n' { printf("Result=%d\n", $1); };
E : E '+' E { $$=$1+$3; }
  | E '-' E { $$=$1-$3; }
  | E '*' E { $$=$1*$3; }
  | E '/' E { $$=$1/$3; }
  | '(' E ')' { $$=$2; }
  | NUM { $$=$1; };
%%
int main(){ yyparse(); return 0; }

Example input:
10+2*5
