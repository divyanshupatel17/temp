LEX YACC 6: Parse Tree Construction (simple expression)

Code (tree.l):
%{
#include "y.tab.h"
%}
%%
[a-zA-Z0-9]+    { yylval.s = strdup(yytext); return ID; }
[+\-*/()]       { return yytext[0]; }
\n              { return '\n'; }
[ \t]           ;
%%
int yywrap(){ return 1; }

Code (tree.y):
%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int yylex(void);
void yyerror(char *s){ printf("Error\n"); }
%}
%union { char *s; }
%token <s> ID
%type <s> E
%%
S : E '\n' { printf("Parse tree (prefix form): %s\n", $1); };
E : E '+' E { asprintf(&$$, "(+ %s %s)", $1, $3); }
  | E '*' E { asprintf(&$$, "(* %s %s)", $1, $3); }
  | '(' E ')' { $$=$2; }
  | ID { $$=$1; };
%%
int main(){ yyparse(); return 0; }

Example input:
a+b*c
