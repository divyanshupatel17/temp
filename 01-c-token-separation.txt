C Program 1: Implementation of Token Separation (Lexical Analysis)

Code (token_sep.c):
#include <stdio.h>
#include <ctype.h>

int main() {
    char s[500];
    int i = 0;
    printf("Enter C statement:\n");
    fgets(s, sizeof(s), stdin);

    while (s[i]) {
        if (isspace((unsigned char)s[i])) { i++; continue; }

        if (isalpha((unsigned char)s[i]) || s[i] == '_') {
            printf("IDENTIFIER: ");
            while (isalnum((unsigned char)s[i]) || s[i] == '_') putchar(s[i++]);
            printf("\n");
        } else if (isdigit((unsigned char)s[i])) {
            printf("NUMBER: ");
            while (isdigit((unsigned char)s[i])) putchar(s[i++]);
            printf("\n");
        } else {
            printf("SYMBOL: %c\n", s[i++]);
        }
    }
    return 0;
}

sum = a1 + 25 * b;
