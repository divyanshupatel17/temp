LEX YACC 3: Infix to Postfix (and note for prefix/interconversion)

Code (infix.l):
%{
#include "y.tab.h"
%}
%%
[a-zA-Z0-9]+  { yylval.str = strdup(yytext); return ID; }
[+\-*/()]     { return yytext[0]; }
\n            { return '\n'; }
[ \t]         ;
%%
int yywrap(){ return 1; }

Code (infix.y):
%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int yylex(void);
void yyerror(char *s){ printf("Invalid\n"); }
%}
%union { char *str; }
%token <str> ID
%type <str> E
%left '+' '-'
%left '*' '/'
%%
S : E '\n' { printf("Postfix: %s\n", $1); };
E : E '+' E { asprintf(&$$, "%s %s +", $1, $3); }
  | E '-' E { asprintf(&$$, "%s %s -", $1, $3); }
  | E '*' E { asprintf(&$$, "%s %s *", $1, $3); }
  | E '/' E { asprintf(&$$, "%s %s /", $1, $3); }
  | '(' E ')' { $$=$2; }
  | ID { $$=$1; };
%%
int main(){ yyparse(); return 0; }

Note:
Prefix can be generated similarly by changing semantic actions.

Example input:
a+b*c

Linux:
yacc -d infix.y
lex infix.l
gcc y.tab.c lex.yy.c -o infix -ll -ly
./infix
