// token_sep.c
#include <stdio.h>
#include <ctype.h>
int main(){
  char s[500]; int i=0;
  fgets(s,sizeof(s),stdin);
  while(s[i]){
    if(isspace((unsigned char)s[i])){ i++; continue; }
    if(isalpha((unsigned char)s[i])||s[i]=='_'){
      printf("ID: "); while(isalnum((unsigned char)s[i])||s[i]=='_') putchar(s[i++]); puts("");
    }else if(isdigit((unsigned char)s[i])){
      printf("NUM: "); while(isdigit((unsigned char)s[i])) putchar(s[i++]); puts("");
    }else printf("SYM: %c\n", s[i++]);
  } return 0;
}
Input:  sum=a1+25*b;
Output (short):
ID: sum
SYM: =
ID: a1
SYM: +
NUM: 25
SYM: *
ID: b
SYM: ;

// symtab.c
#include <stdio.h>
#include <string.h>
#include <ctype.h>
struct Sym{ char n[40], t[10]; int s, a; } st[200];
int cnt=0, addr=1000;
int sz(char *t){ if(!strcmp(t,"char"))return 1; if(!strcmp(t,"double"))return 8; return 4; }
int kw(char *x){ char *k[]={"int","float","double","char","if","else","for","while","return","void","main"}; for(int i=0;i<11;i++) if(!strcmp(x,k[i])) return 1; return 0; }
void add(char *n,char *t){ for(int i=0;i<cnt;i++) if(!strcmp(st[i].n,n)) return; strcpy(st[cnt].n,n); strcpy(st[cnt].t,t); st[cnt].s=sz(t); st[cnt].a=addr; addr+=st[cnt].s; cnt++; }
int main(){
  int L; char line[400], cur[10]=""; scanf("%d",&L); getchar();
  for(int z=0;z<L;z++){ fgets(line,sizeof(line),stdin); int i=0;
    while(line[i]){
      if(isalpha((unsigned char)line[i])||line[i]=='_'){
        char tok[50]; int j=0; while(isalnum((unsigned char)line[i])||line[i]=='_') tok[j++]=line[i++]; tok[j]='\0';
        if(!strcmp(tok,"int")||!strcmp(tok,"float")||!strcmp(tok,"double")||!strcmp(tok,"char")) strcpy(cur,tok);
        else if(cur[0] && !kw(tok)) add(tok,cur);
      } else { if(line[i]==';') cur[0]='\0'; i++; }
    }
  }
  puts("Name\tType\tSize\tAddress");
  for(int i=0;i<cnt;i++) printf("%s\t%s\t%d\t%d\n",st[i].n,st[i].t,st[i].s,st[i].a);
  return 0;
}

double d;
int sum;

b int 4 1004
x float 4 1008
c char 1 1012
d double 8 1013

3) C - Shift-Reduce (LALR-style demo)

// sr.c
#include <stdio.h>
int main(){
  char in[100], st[100]=""; int i=0,t=-1;
  scanf("%s",in); // use i for id
  while(in[i]){
    st[++t]=in[i++]; st[t+1]='\0';
    if(st[t]=='i') st[t]='E';
    if(t>=2 && st[t-2]=='E' && (st[t-1]=='+'||st[t-1]=='*') && st[t]=='E'){ t-=2; st[t]='E'; }
    if(t>=2 && st[t-2]=='(' && st[t-1]=='E' && st[t]==')'){ t-=2; st[t]='E'; }
  }
  puts((t==0 && st[0]=='E')?"Accepted":"Rejected");
  return 0;
}
Input:  i+i*i
Output: Accepted
Linux:   gcc sr.c -o sr && ./sr
Windows: gcc sr.c -o sr.exe && sr.exe
