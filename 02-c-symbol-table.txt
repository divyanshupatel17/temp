C Program 2: Construction of Symbol Table
(take C program from input and show name, type, size, address)

Code (symtab.c):
#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdlib.h>

struct Sym {
    char name[50];
    char type[20];
    int size;
    int addr;
} table[200];

int n = 0, nextAddr = 1000;

int typeSize(const char *t) {
    if (!strcmp(t, "char")) return 1;
    if (!strcmp(t, "int")) return 4;
    if (!strcmp(t, "float")) return 4;
    if (!strcmp(t, "double")) return 8;
    return 4;
}

int isKeyword(const char *s) {
    const char *kw[] = {"int","float","double","char","if","else","for","while","return","void","main"};
    for (int i = 0; i < 11; i++) if (!strcmp(s, kw[i])) return 1;
    return 0;
}

void addSymbol(const char *name, const char *type) {
    for (int i = 0; i < n; i++) if (!strcmp(table[i].name, name)) return;
    strcpy(table[n].name, name);
    strcpy(table[n].type, type);
    table[n].size = typeSize(type);
    table[n].addr = nextAddr;
    nextAddr += table[n].size;
    n++;
}

int main() {
    char line[500], curType[20] = "";
    printf("Enter number of lines: ");
    int lines;
    scanf("%d", &lines);
    getchar();

    for (int l = 0; l < lines; l++) {
        fgets(line, sizeof(line), stdin);
        int i = 0;
        while (line[i]) {
            if (isalpha((unsigned char)line[i]) || line[i] == '_') {
                char tok[60];
                int j = 0;
                while (isalnum((unsigned char)line[i]) || line[i] == '_') tok[j++] = line[i++];
                tok[j] = '\0';

                if (!strcmp(tok, "int") || !strcmp(tok, "float") || !strcmp(tok, "double") || !strcmp(tok, "char")) {
                    strcpy(curType, tok);
                } else if (!isKeyword(tok) && strlen(curType) > 0) {
                    addSymbol(tok, curType);
                }
            } else {
                if (line[i] == ';') curType[0] = '\0';
                i++;
            }
        }
    }

    printf("\nSymbol Table\n");
    printf("Name\tType\tSize\tAddress\n");
    for (int i = 0; i < n; i++) {
        printf("%s\t%s\t%d\t%d\n", table[i].name, table[i].type, table[i].size, table[i].addr);
    }
    return 0;
}

Example input:
5
int a,b;
float x;
char c;
double d;
int sum;

Linux:
gcc symtab.c -o symtab
./symtab

Windows:
gcc symtab.c -o symtab.exe
symtab.exe
