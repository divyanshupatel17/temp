LEX YACC 2: Syntax Checking of Assignment Statements (a=10;)

Code (assign.l):
%{
#include "y.tab.h"
%}
%%
[a-zA-Z_][a-zA-Z0-9_]*  { return ID; }
[0-9]+                  { return NUM; }
"="                     { return '='; }
";"                     { return ';'; }
[ \t\n]+                ;
.                       { return yytext[0]; }
%%
int yywrap(){ return 1; }

Code (assign.y):
%{
#include <stdio.h>
int yylex(void);
void yyerror(char *s){ printf("Invalid assignment\n"); }
%}
%token ID NUM
%%
S : ID '=' NUM ';' { printf("Valid assignment\n"); };
%%
int main(){ yyparse(); return 0; }

Example input:
a=10;

Linux:
yacc -d assign.y
lex assign.l
gcc y.tab.c lex.yy.c -o assign -ll -ly
./assign

Windows:
win_bison -d assign.y
win_flex assign.l
gcc assign.tab.c lex.yy.c -o assign.exe
assign.exe
