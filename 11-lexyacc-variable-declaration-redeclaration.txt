LEX YACC 4: Variable Declaration Parsing with Redeclaration Detection

Code (decl.l):
%{
#include "y.tab.h"
%}
%%
"int"|"float"|"char"      { return TYPE; }
[a-zA-Z_][a-zA-Z0-9_]*    { yylval.s = strdup(yytext); return ID; }
","                       { return ','; }
";"                       { return ';'; }
[ \t\n]+                  ;
.                         { return yytext[0]; }
%%
int yywrap(){ return 1; }

Code (decl.y):
%{
#include <stdio.h>
#include <string.h>
char sym[100][50]; int n=0,err=0;
int yylex(void);
void yyerror(char *s){ err=1; }
int exists(char *x){ for(int i=0;i<n;i++) if(!strcmp(sym[i],x)) return 1; return 0; }
void add(char *x){ strcpy(sym[n++],x); }
%}
%union { char *s; }
%token TYPE
%token <s> ID
%%
S : TYPE L ';' { if(!err) printf("Valid declaration\n"); };
L : L ',' ID { if(exists($3)){ printf("Redeclaration: %s\n",$3); err=1; } else add($3); }
  | ID       { if(exists($1)){ printf("Redeclaration: %s\n",$1); err=1; } else add($1); };
%%
int main(){ yyparse(); return 0; }

Example input:
int a,b,a;
