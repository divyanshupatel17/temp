LEX YACC 9: Generate Three-Address Code for Expression

Code (tac.l):
%{
#include "y.tab.h"
%}
%%
[a-zA-Z0-9]+  { yylval.s = strdup(yytext); return ID; }
[+\-*/()]     { return yytext[0]; }
\n            { return '\n'; }
[ \t]         ;
%%
int yywrap(){ return 1; }

Code (tac.y):
%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int yylex(void), t=1;
void yyerror(char *s){ printf("Invalid\n"); }
%}
%union { char *s; }
%token <s> ID
%type <s> E
%%
S : E '\n' { printf("Final: %s\n", $1); };
E : E '+' E { char *r=(char*)malloc(10); sprintf(r,"t%d",t++); printf("%s = %s + %s\n",r,$1,$3); $$=r; }
  | E '-' E { char *r=(char*)malloc(10); sprintf(r,"t%d",t++); printf("%s = %s - %s\n",r,$1,$3); $$=r; }
  | E '*' E { char *r=(char*)malloc(10); sprintf(r,"t%d",t++); printf("%s = %s * %s\n",r,$1,$3); $$=r; }
  | E '/' E { char *r=(char*)malloc(10); sprintf(r,"t%d",t++); printf("%s = %s / %s\n",r,$1,$3); $$=r; }
  | '(' E ')' { $$=$2; }
  | ID { $$=$1; };
%%
int main(){ yyparse(); return 0; }

Example input:
a+b*c

Linux:
yacc -d tac.y
lex tac.l
gcc y.tab.c lex.yy.c -o tac -ll -ly
./tac
